#ЗАДАЧА 2
import sys
import math

print("\n=== Предельные значения числовых типов ===")

print("\n--- Целые числа (int) ---")
print(f"Минимальное значение int: {-sys.maxsize - 1}")
print(f"Максимальное значение int: {sys.maxsize}")
print("Примечание: В Python 3 целые числа (int) имеют произвольную точность,")
print("но sys.maxsize показывает максимальное значение, которое можно хранить")
print("как обычное целое число (не 'длинное').")

print("\n--- Числа с плавающей точкой (float) ---")
print(f"Миниманое положительное нормализованное float: {sys.float_info.min}")
print(f"Максимальное значение float: {sys.float_info.max}")
print(f"Миниманое положительное денормализованное float: {sys.float_info.min * sys.float_info.epsilon}")
print(f"Эпсилон (машинный ноль): {sys.float_info.epsilon}")
print("Примечание: Float в Python - это 64-битные числа с плавающей точкой")
print("стандарта IEEE 754. Миниманое значение отрицательное такое же,")
print("но со знаком минус: -sys.float_info.max")

print("\n--- Комплексные числа (complex) ---")
print("Комплексные числа состоят из двух float (действительная и мнимая части),")
print("поэтому их пределы такие же, как у float.")
print(f"Миниманое значение компоненты: {sys.float_info.min}")
print(f"Максимальное значение компоненты: {sys.float_info.max}")

#ЗАДАЧА 3
print("\n=== Калькулятор объема икосаэдра ===")


try:
    a = float(input("Введите длину ребра икосаэдра: "))

    if a <= 0:
        print("Ошибка: Длина ребра должна быть положительным числом!")
    else:
        volume = (5 * (3 + math.sqrt(5)) / 12) * (a ** 3)
        print(f"Длина ребра икосаэдра: {a}")
        print(f"Объем икосаэдра: {volume:.4f}")
except ValueError:
    print("Ошибка: Введите корректное число!")
